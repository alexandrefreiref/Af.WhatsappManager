Management of sending text via URL

# Install
npm install @aninga/whatsapp-manager 

# Usage
import {Send} from '@aninga/whatsapp-manager'
or
import * as WA from '@aninga/whatsapp-manager'

WA.Send('5569999999999', 'Hello')
